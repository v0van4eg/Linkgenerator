def safe_folder_name(name: str) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç —Å—Ç—Ä–æ–∫—É –≤ –±–µ–∑–æ–ø–∞—Å–Ω–æ–µ –∏–º—è –ø–∞–ø–∫–∏:
    - –û—Å—Ç–∞–≤–ª—è–µ—Ç –±—É–∫–≤—ã (–≤–∫–ª—é—á–∞—è –∫–∏—Ä–∏–ª–ª–∏—Ü—É), —Ü–∏—Ñ—Ä—ã, –¥–µ—Ñ–∏—Å, –ø–æ–¥—á—ë—Ä–∫–∏–≤–∞–Ω–∏–µ
    - –ó–∞–º–µ–Ω—è–µ—Ç –ø—Ä–æ–±–µ–ª—ã –∏ –¥—Ä—É–≥–∏–µ —Ä–∞–∑–¥–µ–ª–∏—Ç–µ–ª–∏ –Ω–∞ –¥–µ—Ñ–∏—Å
    - –£–¥–∞–ª—è–µ—Ç –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–æ –æ–ø–∞—Å–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã
    - –£–±–∏—Ä–∞–µ—Ç –Ω–∞—á–∞–ª—å–Ω—ã–µ/–∫–æ–Ω–µ—á–Ω—ã–µ –¥–µ—Ñ–∏—Å—ã –∏ —Ç–æ—á–∫–∏
    """
    if not name:
        return "unnamed"
    # –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º Unicode (–Ω–∞–ø—Ä–∏–º–µ—Ä, –±—É–∫–≤—ã —Å –¥–∏–∞–∫—Ä–∏—Ç–∏–∫–æ–π)
    name = unicodedata.normalize('NFKD', name)
    # –†–∞–∑—Ä–µ—à—ë–Ω–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã: –±—É–∫–≤—ã (–≤—Å–µ –∞–ª—Ñ–∞–≤–∏—Ç—ã), —Ü–∏—Ñ—Ä—ã, –¥–µ—Ñ–∏—Å, –ø–æ–¥—á—ë—Ä–∫–∏–≤–∞–Ω–∏–µ
    # –ó–∞–º–µ–Ω—è–µ–º –≤—Å—ë –æ—Å—Ç–∞–ª—å–Ω–æ–µ –Ω–∞ –¥–µ—Ñ–∏—Å
    name = re.sub(r'[^\w\s-]', '', name, flags=re.UNICODE)
    name = re.sub(r'[-\s]+', '-', name, flags=re.UNICODE).strip('-_')
    # –û–≥—Ä–∞–Ω–∏—á–∏–º –¥–ª–∏–Ω—É, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø—Ä–æ–±–ª–µ–º —Å –û–°
    name = name[:255] if name else "unnamed"
    return name


def process_zip_archive(zip_file, client_name):
    """
    –û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç ZIP-–∞—Ä—Ö–∏–≤ –∏ –∏–∑–≤–ª–µ–∫–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –≤ —Å—Ç—Ä—É–∫—Ç—É—Ä—É –∫–∞—Ç–∞–ª–æ–≥–æ–≤ –∫–ª–∏–µ–Ω—Ç–∞
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ URL —Å–æ–∑–¥–∞–Ω–Ω—ã—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π
    """
    image_urls = []

    # –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—É—é –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é –¥–ª—è —Ä–∞—Å–ø–∞–∫–æ–≤–∫–∏
    with tempfile.TemporaryDirectory() as temp_dir:
        # –°–æ—Ö—Ä–∞–Ω—è–µ–º ZIP —Ñ–∞–π–ª
        zip_path = os.path.join(temp_dir, zip_file.filename)
        zip_file.save(zip_path)

        # –†–∞—Å–ø–∞–∫–æ–≤—ã–≤–∞–µ–º –∞—Ä—Ö–∏–≤
        with zipfile.ZipFile(zip_path, 'r') as zip_ref:
            zip_ref.extractall(temp_dir)

        # –û–±—Ö–æ–¥–∏–º –≤—Å–µ —Ñ–∞–π–ª—ã –≤ —Ä–∞—Å–ø–∞–∫–æ–≤–∞–Ω–Ω–æ–π –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏–∏
        for root, dirs, files in os.walk(temp_dir):
            for file in files:
                # –ü—Ä–æ–ø—É—Å–∫–∞–µ–º —Å–∏—Å—Ç–µ–º–Ω—ã–µ —Ñ–∞–π–ª—ã
                if file.lower() in ['thumbs.db', '.ds_store']:
                    continue

                # –ü—Ä–æ–≤–µ—Ä—è–µ–º —Ä–∞—Å—à–∏—Ä–µ–Ω–∏–µ —Ñ–∞–π–ª–∞
                if not allowed_file(file):
                    continue

                # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∞—Ä—Ç–∏–∫—É–ª –∏–∑ –∏–º–µ–Ω–∏ –ø–∞–ø–∫–∏
                relative_path = os.path.relpath(root, temp_dir)
                if relative_path == '.':
                    # –§–∞–π–ª—ã –≤ –∫–æ—Ä–Ω–µ –∞—Ä—Ö–∏–≤–∞ - –ø—Ä–æ–ø—É—Å–∫–∞–µ–º
                    continue

                # –ü–µ—Ä–≤–∞—è –ø–∞–ø–∫–∞ –≤ –ø—É—Ç–∏ - —ç—Ç–æ –∞—Ä—Ç–∏–∫—É–ª
                article = os.path.basename(root) if relative_path.count(os.sep) == 0 else relative_path.split(os.sep)[0]

                # –°–æ–∑–¥–∞–µ–º –ø—É—Ç–∏ –¥–ª—è –∫–ª–∏–µ–Ω—Ç–∞ –∏ –∞—Ä—Ç–∏–∫—É–ª–∞
                client_folder = safe_folder_name(client_name)
                article_folder = safe_folder_name(article)

                full_path = os.path.join(Config.UPLOAD_FOLDER, client_folder, article_folder)
                os.makedirs(full_path, exist_ok=True)

                # –ö–æ–ø–∏—Ä—É–µ–º —Ñ–∞–π–ª –≤ —Ü–µ–ª–µ–≤—É—é –¥–∏—Ä–µ–∫—Ç–æ—Ä–∏—é
                source_file = os.path.join(root, file)
                target_file = os.path.join(full_path, file)

                shutil.copy2(source_file, target_file)

                # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º URL
                image_url = "{}/images/{}/{}/{}".format(
                    Config.BASE_URL,
                    quote(client_folder, safe=''),
                    quote(article_folder, safe=''),
                    quote(file, safe='')
                )
                image_urls.append({
                    'url': image_url,
                    'article': article,
                    'filename': file
                })

    return image_urls


def generate_megamarket_xlsx(image_data, client_name):
    """
    –ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç XLSX –¥–æ–∫—É–º–µ–Ω—Ç –¥–ª—è –ú–µ–≥–∞–º–∞—Ä–∫–µ—Ç –ø–æ —à–∞–±–ª–æ–Ω—É
    –°—Ç–æ–ª–±—Ü—ã: "–ö–æ–¥ —Ç–æ–≤–∞—Ä–∞ –°–ú–ú(–æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ)", "–°—Å—ã–ª–∫–∞ –Ω–∞ –æ—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ", "–°—Å—ã–ª–∫–∞ –Ω–∞ –¥–æ–ø. —Ñ–æ—Ç–æ ‚Ññ1" - "–°—Å—ã–ª–∫–∞ –Ω–∞ –¥–æ–ø. —Ñ–æ—Ç–æ ‚Ññ9"
    """
    try:
        # –ü—ã—Ç–∞–µ–º—Å—è –∑–∞–≥—Ä—É–∑–∏—Ç—å —à–∞–±–ª–æ–Ω, –µ—Å–ª–∏ –æ–Ω —Å—É—â–µ—Å—Ç–≤—É–µ—Ç
        template_path = os.path.join('templates', 'megamarket.xlsx')
        if os.path.exists(template_path):
            wb = load_workbook(template_path)
            ws = wb.active
            # –ù–∞—á–∏–Ω–∞–µ–º —Å–æ –≤—Ç–æ—Ä–æ–π —Å—Ç—Ä–æ–∫–∏ (–ø–æ—Å–ª–µ –∑–∞–≥–æ–ª–æ–≤–∫–æ–≤)
            start_row = 2
        else:
            # –°–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π —Ñ–∞–π–ª –µ—Å–ª–∏ —à–∞–±–ª–æ–Ω –Ω–µ –Ω–∞–π–¥–µ–Ω
            wb = Workbook()
            ws = wb.active
            ws.title = "Megamarket Images"

            # –ó–∞–≥–æ–ª–æ–≤–∫–∏ —Å—Ç–æ–ª–±—Ü–æ–≤
            headers = ["–ö–æ–¥ —Ç–æ–≤–∞—Ä–∞ –°–ú–ú(–æ–±—è–∑–∞—Ç–µ–ª—å–Ω–æ)", "–°—Å—ã–ª–∫–∞ –Ω–∞ –æ—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ"]
            for i in range(1, 10):
                headers.append(f"–°—Å—ã–ª–∫–∞ –Ω–∞ –¥–æ–ø. —Ñ–æ—Ç–æ ‚Ññ{i}")

            ws.append(headers)
            start_row = 2

            # –°—Ç–∏–ª–∏ –¥–ª—è –∑–∞–≥–æ–ª–æ–≤–∫–æ–≤
            header_font = Font(bold=True)
            for cell in ws[1]:
                cell.font = header_font
                cell.alignment = Alignment(horizontal='center', vertical='center')

        # –ì—Ä—É–ø–ø–∏—Ä—É–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –ø–æ –∞—Ä—Ç–∏–∫—É–ª–∞–º
        articles = {}
        for item in image_data:
            article = item['article']
            if article not in articles:
                articles[article] = []
            articles[article].append(item['url'])

        print(f"–ú–µ–≥–∞–º–∞—Ä–∫–µ—Ç: –Ω–∞–π–¥–µ–Ω–æ –∞—Ä—Ç–∏–∫—É–ª–æ–≤: {len(articles)}")

        # –°–æ–∑–¥–∞–µ–º —Å—Ç—Ä–æ–∫–∏ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –∞—Ä—Ç–∏–∫—É–ª–∞
        current_row = start_row
        for article, urls in articles.items():
            # –û—Å–Ω–æ–≤–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
            row_data = [article]  # –ö–æ–¥ —Ç–æ–≤–∞—Ä–∞ –°–ú–ú

            # –î–æ–±–∞–≤–ª—è–µ–º —Å—Å—ã–ª–∫–∏ –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
            # –ü–µ—Ä–≤–∞—è —Å—Å—ã–ª–∫–∞ - –æ—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ
            if urls:
                row_data.append(urls[0])  # –û—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ
                # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ —Ñ–æ—Ç–æ (–º–∞–∫—Å–∏–º—É–º 9)
                for i in range(1, 10):
                    if i < len(urls):
                        row_data.append(urls[i])
                    else:
                        row_data.append("")
            else:
                # –ï—Å–ª–∏ –Ω–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π, –¥–æ–±–∞–≤–ª—è–µ–º –ø—É—Å—Ç—ã–µ —è—á–µ–π–∫–∏
                row_data.append("")  # –û—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ
                for i in range(1, 10):
                    row_data.append("")

            # –ó–∞–ø–∏—Å—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ
            for col, value in enumerate(row_data, 1):
                ws.cell(row=current_row, column=col, value=value)

            current_row += 1

        # –ù–∞—Å—Ç—Ä–∞–∏–≤–∞–µ–º —à–∏—Ä–∏–Ω—É —Å—Ç–æ–ª–±—Ü–æ–≤
        column_widths = {
            'A': 20,  # –ö–æ–¥ —Ç–æ–≤–∞—Ä–∞
            'B': 40,  # –û—Å–Ω–æ–≤–Ω–æ–µ —Ñ–æ—Ç–æ
        }
        for i in range(3, 12):  # –°—Ç–æ–ª–±—Ü—ã C-K –¥–ª—è –¥–æ–ø. —Ñ–æ—Ç–æ
            column_letter = get_column_letter(i)
            column_widths[column_letter] = 40

        for col, width in column_widths.items():
            ws.column_dimensions[col].width = width

        # –°–æ—Ö—Ä–∞–Ω—è–µ–º –≤ –±—É—Ñ–µ—Ä
        buffer = io.BytesIO()
        wb.save(buffer)
        buffer.seek(0)

        return buffer

    except Exception as e:
        app.logger.error(f"Error generating Megamarket XLSX: {str(e)}")
        raise

            # –î–æ–±–∞–≤–ª—è–µ–º —Å—Å—ã–ª–∫–∏ –Ω–∞ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è
            max_images = 20
            for i, url in enumerate(urls):
                if i >= max_images:
                    break
                row_data.append(url)

            # –ó–∞–ø–æ–ª–Ω—è–µ–º –æ—Å—Ç–∞–≤—à–∏–µ—Å—è —è—á–µ–π–∫–∏ –¥–ª—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –ø—É—Å—Ç—ã–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏
            remaining_images = max_images - min(len(urls), max_images)
            for _ in range(remaining_images):
                row_data.append("")

            # –ó–∞–ø–∏—Å—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ
            for col, value in enumerate(row_data, 1):
                ws.cell(row=current_row, column=col, value=value)

            current_row += 1

        # –ù–∞—Å—Ç—Ä–∞–∏–≤–∞–µ–º —à–∏—Ä–∏–Ω—É —Å—Ç–æ–ª–±—Ü–æ–≤
        column_widths = {
            'A': 15, 'B': 30, 'C': 15, 'D': 20, 'E': 15,
            'F': 25, 'G': 25, 'H': 40, 'I': 20, 'J': 20,
            'K': 25
        }

        # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —à–∏—Ä–∏–Ω—É –¥–ª—è —Å—Ç–æ–ª–±—Ü–æ–≤ —Å –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º–∏
        for i in range(12, 12 + 20):  # –°—Ç–æ–ª–±—Ü—ã L-AE –¥–ª—è –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π
            column_letter = get_column_letter(i)
            column_widths[column_letter] = 30

        for col, width in column_widths.items():
            ws.column_dimensions[col].width = width

        # –°–æ—Ö—Ä–∞–Ω—è–µ–º –≤ –±—É—Ñ–µ—Ä
        buffer = io.BytesIO()
        wb.save(buffer)
        buffer.seek(0)

        return buffer

    except Exception as e:
        app.logger.error(f"Error generating ELISE XLSX: {str(e)}")
        raise

def handle_single_upload(request):
    """–û–±—Ä–∞–±–æ—Ç–∫–∞ –∑–∞–≥—Ä—É–∑–∫–∏ –æ—Ç–¥–µ–ª—å–Ω—ã—Ö –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π"""
    client_name = request.form.get('client_name', '').strip()
    product_name = request.form.get('product_name', '').strip()

    if not client_name or not product_name:
        return render_template('index.html',
                               error='–ó–∞–ø–æ–ª–Ω–∏—Ç–µ –≤—Å–µ –ø–æ–ª—è',
                               clients=Config.CLIENTS,
                               selected_client=client_name,
                               product_name=product_name)

    # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –≤—ã–±—Ä–∞–Ω–Ω—ã–π –∫–ª–∏–µ–Ω—Ç –µ—Å—Ç—å –≤ —Å–ø–∏—Å–∫–µ
    if client_name not in Config.CLIENTS:
        return render_template('index.html',
                               error='–í—ã–±–µ—Ä–∏—Ç–µ –∫–ª–∏–µ–Ω—Ç–∞ –∏–∑ —Å–ø–∏—Å–∫–∞',
                               clients=Config.CLIENTS,
                               selected_client=client_name,
                               product_name=product_name)

    # –°–æ–∑–¥–∞–µ–º –ø—É—Ç–∏ –¥–ª—è –∫–ª–∏–µ–Ω—Ç–∞ –∏ —Ç–æ–≤–∞—Ä–∞
    client_folder = safe_folder_name(client_name)
    product_folder = safe_folder_name(product_name)

    full_path = os.path.join(Config.UPLOAD_FOLDER, client_folder, product_folder)
    os.makedirs(full_path, exist_ok=True)

    uploaded_files = request.files.getlist('images')
    image_urls = []

    for file in uploaded_files:
        if file and allowed_file(file.filename):
            # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —É–Ω–∏–∫–∞–ª—å–Ω–æ–µ –∏–º—è —Ñ–∞–π–ª–∞
            random_hex = uuid.uuid4().hex[:6]
            file_extension = os.path.splitext(file.filename)[1]
            file_name = os.path.splitext(file.filename)[0]
            unique_filename = f"{file_name}-{random_hex}{file_extension}"

            # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ñ–∞–π–ª
            file_path = os.path.join(full_path, unique_filename)
            file.save(file_path)

            # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º URL
            image_url = "{}/images/{}/{}/{}".format(
                Config.BASE_URL,
                quote(client_folder, safe=''),
                quote(product_folder, safe=''),
                quote(unique_filename, safe='')
            )
            image_urls.append({
                'url': image_url,
                'article': product_name,
                'filename': unique_filename
            })

    return render_template('index.html',
                           image_urls=image_urls,
                           clients=Config.CLIENTS,
                           selected_client=client_name,
                           product_name=product_name)

@app.route('/download-links')

    urls = request.args.getlist('urls')

        return "No URLs provided", 400

    # –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
    temp_file = f"temp_links_{uuid.uuid4().hex}.txt"
    with open(temp_file, 'w', encoding='utf-8') as f:
        f.write('\n'.join(urls))

    return send_file(temp_file,
                     as_attachment=True,
                     download_name='image_links.txt',
                     mimetype='text/plain')


@app.route('/download-xlsx', methods=['POST'])
def download_xlsx():
    try:
        data = request.get_json()
        if not data:
            return jsonify({'error': 'No data provided'}), 400

        image_data = data.get('image_data', [])
        client_name = data.get('client_name', '')

        if not image_data:
            return jsonify({'error': 'No image data provided'}), 400

        print(f"–ü–æ–ª—É—á–µ–Ω–æ –¥–∞–Ω–Ω—ã—Ö –¥–ª—è XLSX: {len(image_data)} —ç–ª–µ–º–µ–Ω—Ç–æ–≤")
        print(f"–ö–ª–∏–µ–Ω—Ç: {client_name}")

        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º XLSX –¥–æ–∫—É–º–µ–Ω—Ç
        xlsx_buffer = generate_xlsx_document(image_data, client_name)

        # –°–æ–∑–¥–∞–µ–º –∏–º—è —Ñ–∞–π–ª–∞
        filename = f"{safe_folder_name(client_name)}_images.xlsx"

        # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ñ–∞–π–ª –≤—Ä–µ–º–µ–Ω–Ω–æ
        temp_file = f"temp_xlsx_{uuid.uuid4().hex}.xlsx"
        with open(temp_file, 'wb') as f:
            f.write(xlsx_buffer.getvalue())

        response = send_file(
            temp_file,
            as_attachment=True,
            download_name=filename,
            mimetype='application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        )

        # –£–¥–∞–ª—è–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª –ø–æ—Å–ª–µ –æ—Ç–ø—Ä–∞–≤–∫–∏
        @response.call_on_close
        def remove_file():
            try:
                os.remove(temp_file)
            except:
                pass
        return response

    except Exception as e:
        app.logger.error(f"Error generating XLSX: {str(e)}")
        return jsonify({'error': f'–û—à–∏–±–∫–∞ –ø—Ä–∏ –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ XLSX-—Ñ–∞–π–ª–∞: {str(e)}'}), 500

                    // –°–æ–∑–¥–∞–µ–º —Å—Å—ã–ª–∫—É –¥–ª—è —Å–∫–∞—á–∏–≤–∞–Ω–∏—è
                    const url = window.URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.style.display = 'none';
                    a.href = url;
                    a.download = `${clientName || 'images'}_catalog.xlsx`;
                    document.body.appendChild(a);
                    a.click();
                    window.URL.revokeObjectURL(url);
                    document.body.removeChild(a);

                    showNotification('XLSX –¥–æ–∫—É–º–µ–Ω—Ç —É—Å–ø–µ—à–Ω–æ —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω –∏ —Å–∫–∞—á–∞–Ω!', 'success');
                })
                .catch(error => {
                    console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ XLSX:', error);
                    showNotification('–û—à–∏–±–∫–∞ –ø—Ä–∏ –≥–µ–Ω–µ—Ä–∞—Ü–∏–∏ XLSX: ' + error.message, 'error');
                });
            }
            // --- –ü—Ä–∏–≤—è–∑–∫–∞ —Å–æ–±—ã—Ç–∏–π –¥–ª—è –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è –ø—Ä–∏ –∫–ª–∏–∫–µ –Ω–∞ —Å—Å—ã–ª–∫—É ---
            document.querySelectorAll('.url-text').forEach(urlElement => {
                urlElement.addEventListener('click', function(e) {
                    // –ù–µ –∫–æ–ø–∏—Ä—É–µ–º –µ—Å–ª–∏ –∫–ª–∏–∫–Ω—É–ª–∏ –Ω–∞ –ø–æ–¥—Å–∫–∞–∑–∫—É
                    if (e.target.classList.contains('copy-hint')) return;

                    const url = this.getAttribute('data-url');
                    if (url) {
                        copyToClipboard(url);

                        // –í–∏–∑—É–∞–ª—å–Ω–∞—è –æ–±—Ä–∞—Ç–Ω–∞—è —Å–≤—è–∑—å
                        this.classList.add('copied');
                        setTimeout(() => {
                            this.classList.remove('copied');
                        }, 2000);
                    }
                });
            });

            // --- –ü—Ä–∏–≤—è–∑–∫–∞ –∫ –∫–Ω–æ–ø–∫–µ "–ö–æ–ø–∏—Ä–æ–≤–∞—Ç—å –≤—Å–µ" ---
            const copyAllBtn = document.getElementById('copyAllBtn');
            if (copyAllBtn) {
                copyAllBtn.addEventListener('click', copyAllToClipboard);
            }

            // --- –ü—Ä–∏–≤—è–∑–∫–∞ –∫ –∫–Ω–æ–ø–∫–µ "–°–∫–∞—á–∞—Ç—å –≤—Å–µ" ---
            const downloadAllBtn = document.getElementById('downloadAllBtn');
            if (downloadAllBtn) {
                downloadAllBtn.addEventListener('click', downloadAllLinks);
            }

            // --- –ü—Ä–∏–≤—è–∑–∫–∞ –∫ –∫–Ω–æ–ø–∫–µ "–°–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å –¥–æ–∫—É–º–µ–Ω—Ç XLSX" ---
            const downloadXLSXBtn = document.getElementById('downloadXLSXBtn');
            if (downloadXLSXBtn) {
                downloadXLSXBtn.addEventListener('click', downloadXLSXDocument);
            }

            // --- –ü—Ä–µ–¥–ø—Ä–æ—Å–º–æ—Ç—Ä —Ñ–∞–π–ª–æ–≤ ---
            const imagesInput = document.getElementById('images');
            if (imagesInput) {
                imagesInput.addEventListener('change', function(e) {
                    const files = e.target.files;
                    let previewContainer = document.querySelector('.upload-preview');

                    // –£–¥–∞–ª—è–µ–º —Å—Ç–∞—Ä—ã–π –ø—Ä–µ–≤—å—é –µ—Å–ª–∏ –µ—Å—Ç—å
                    if (previewContainer) {
                        previewContainer.remove();
                    }

                    if (files.length > 0) {
                        previewContainer = document.createElement('div');
                        previewContainer.className = 'upload-preview';
                        previewContainer.innerHTML = '<h3>–í—ã–±—Ä–∞–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã:</h3>';

                        Array.from(files).forEach(file => {
                            if (file.type.startsWith('image/')) {
                                const reader = new FileReader();
                                reader.onload = function(e) {
                                    const img = document.createElement('img');
                                    img.src = e.target.result;
                                    img.style.cssText = 'max-width: 100px; max-height: 100px; margin: 5px;';
                                    previewContainer.appendChild(img);
                                };
                                reader.readAsDataURL(file);
                            }
                        });

                        document.querySelector('.upload-section').appendChild(previewContainer);
                    }
                });
            }
        });
