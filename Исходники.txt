def safe_folder_name(name: str) -> str:
    """
    –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç —Å—Ç—Ä–æ–∫—É –≤ –±–µ–∑–æ–ø–∞—Å–Ω–æ–µ –∏–º—è –ø–∞–ø–∫–∏:
    - –û—Å—Ç–∞–≤–ª—è–µ—Ç –±—É–∫–≤—ã (–≤–∫–ª—é—á–∞—è –∫–∏—Ä–∏–ª–ª–∏—Ü—É), —Ü–∏—Ñ—Ä—ã, –¥–µ—Ñ–∏—Å, –ø–æ–¥—á—ë—Ä–∫–∏–≤–∞–Ω–∏–µ
    - –ó–∞–º–µ–Ω—è–µ—Ç –ø—Ä–æ–±–µ–ª—ã –∏ –¥—Ä—É–≥–∏–µ —Ä–∞–∑–¥–µ–ª–∏—Ç–µ–ª–∏ –Ω–∞ –¥–µ—Ñ–∏—Å
    - –£–¥–∞–ª—è–µ—Ç –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–æ –æ–ø–∞—Å–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã
    - –£–±–∏—Ä–∞–µ—Ç –Ω–∞—á–∞–ª—å–Ω—ã–µ/–∫–æ–Ω–µ—á–Ω—ã–µ –¥–µ—Ñ–∏—Å—ã –∏ —Ç–æ—á–∫–∏
    """
    if not name:
        return "unnamed"
    # –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º Unicode (–Ω–∞–ø—Ä–∏–º–µ—Ä, –±—É–∫–≤—ã —Å –¥–∏–∞–∫—Ä–∏—Ç–∏–∫–æ–π)
    name = unicodedata.normalize('NFKD', name)
    # –†–∞–∑—Ä–µ—à—ë–Ω–Ω—ã–µ —Å–∏–º–≤–æ–ª—ã: –±—É–∫–≤—ã (–≤—Å–µ –∞–ª—Ñ–∞–≤–∏—Ç—ã), —Ü–∏—Ñ—Ä—ã, –¥–µ—Ñ–∏—Å, –ø–æ–¥—á—ë—Ä–∫–∏–≤–∞–Ω–∏–µ
    # –ó–∞–º–µ–Ω—è–µ–º –≤—Å—ë –æ—Å—Ç–∞–ª—å–Ω–æ–µ –Ω–∞ –¥–µ—Ñ–∏—Å
    name = re.sub(r'[^\w\s-]', '', name, flags=re.UNICODE)
    name = re.sub(r'[-\s]+', '-', name, flags=re.UNICODE).strip('-_')
    # –û–≥—Ä–∞–Ω–∏—á–∏–º –¥–ª–∏–Ω—É, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –ø—Ä–æ–±–ª–µ–º —Å –û–°
    name = name[:255] if name else "unnamed"
    return name

@app.route('/', methods=['GET', 'POST'])
def index():
    if request.method == 'POST':
        # –ü–æ–ª—É—á–∞–µ–º –¥–∞–Ω–Ω—ã–µ –∏–∑ —Ñ–æ—Ä–º—ã
        client_name = request.form.get('client_name', '').strip()
        product_name = request.form.get('product_name', '').strip()

        if not client_name or not product_name:
            return render_template('index.html', error='–ó–∞–ø–æ–ª–Ω–∏—Ç–µ –≤—Å–µ –ø–æ–ª—è')

        # –°–æ–∑–¥–∞–µ–º –ø—É—Ç–∏ –¥–ª—è –∫–ª–∏–µ–Ω—Ç–∞ –∏ —Ç–æ–≤–∞—Ä–∞
        client_folder = safe_folder_name(client_name)
        product_folder = safe_folder_name(product_name)

        full_path = os.path.join(Config.UPLOAD_FOLDER, client_folder, product_folder)
        os.makedirs(full_path, exist_ok=True)

        uploaded_files = request.files.getlist('images')
        image_urls = []

        for file in uploaded_files:
            if file and allowed_file(file.filename):
                # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —É–Ω–∏–∫–∞–ª—å–Ω–æ–µ –∏–º—è —Ñ–∞–π–ª–∞
                random_hex = uuid.uuid4().hex[:6]
                file_extension = os.path.splitext(file.filename)[1]
                file_name = os.path.splitext(file.filename)[0]
                unique_filename = f"{file_name}-{random_hex}{file_extension}"

                # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ñ–∞–π–ª
                file_path = os.path.join(full_path, unique_filename)
                file.save(file_path)

                # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º URL
                # image_url = f"{Config.BASE_URL}/images/{client_folder}/{product_folder}/{unique_filename}"
                image_url = "{}/images/{}/{}/{}".format(
                    Config.BASE_URL,
                    quote(client_folder, safe=''),
                    quote(product_folder, safe=''),
                    quote(unique_filename, safe='')
                )
                image_urls.append(image_url)

        return render_template('index.html',
                               image_urls=image_urls,
                               client_name=client_name,
                               product_name=product_name)

    return render_template('index.html')


@app.route('/download-links')
def download_links():
    urls = request.args.getlist('urls')
    if not urls:
        return "No URLs provided", 400

    # –°–æ–∑–¥–∞–µ–º –≤—Ä–µ–º–µ–Ω–Ω—ã–π —Ñ–∞–π–ª
    temp_file = f"temp_links_{uuid.uuid4().hex}.txt"
    with open(temp_file, 'w', encoding='utf-8') as f:
        f.write('\n'.join(urls))

    return send_file(temp_file,
                     as_attachment=True,
                     download_name='image_links.txt',
                     mimetype='text/plain')
